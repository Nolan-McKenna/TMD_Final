3) Aggregate Model Scores¶

Higher is better for:

  • coverage_score
  • keyword_overlap
  • similarity
  • faithfulness_score

Lower is better for:

  • contradiction_rate
  • abs_sentiment_shift (smaller sentiment drift)

weights = {
    'coverage_score': 0.35,
    'faithfulness_score': 0.35,
    'keyword_overlap': 0.15,
    'similarity': 0.10,
    'contradiction_rate': 0.05,  # subtract
}

rank_df = model_summary.copy()
rank_df['composite_score'] = (
    weights['coverage_score'] * rank_df['coverage_score']
    + weights['faithfulness_score'] * rank_df['faithfulness_score']
    + weights['keyword_overlap'] * rank_df['keyword_overlap']
    + weights['similarity'] * rank_df['similarity']
    - weights['contradiction_rate'] * rank_df['contradiction_rate']
)

rank_df = rank_df.sort_values('composite_score', ascending=False)

def print_run_summary(df):
    print('=== RUN SUMMARY ===')
    print('Rows:', len(df))
    print('Articles:', df['Article Name'].nunique())
    print('Genres:', sorted(df['Genre'].dropna().unique().tolist()))
    print('Models:', sorted(df['Model'].dropna().unique().tolist()))

    cols = ['coverage_score', 'faithfulness_score', 'keyword_overlap', 'similarity', 'contradiction_rate']
    available = [c for c in cols if c in df.columns]
    print('\n=== OVERALL MODEL MEANS ===')
    display(df.groupby('Model', dropna=False)[available].mean().round(4).sort_index())

def print_article_summary(df, article_name):
    sub = df[df['Article Name'].str.lower() == article_name.strip().lower()].copy()
    if sub.empty:
        print(f'No rows for article: {article_name}')
        return
    cols = ['Genre', 'Article Name', 'Model', 'coverage_score', 'faithfulness_score', 'keyword_overlap', 'similarity', 'contradiction_rate', 'framing_label']
    cols = [c for c in cols if c in sub.columns]
    print(f'=== ARTICLE SUMMARY: {article_name} ===')
    display(sub[cols].sort_values('Model'))

print_run_summary(df)

import numpy as np

def annotate_bars(ax, fmt='{:.3f}', dy=0.01):
    for bar in ax.patches:
        h = bar.get_height()
        if h is None or np.isnan(h):
            continue
        ax.text(bar.get_x() + bar.get_width()/2, h + dy, fmt.format(h), ha='center', va='bottom', fontsize=9)

def model_metric_bar(df, metric, title=None, ylim=(0,1), ascending=False):
    s = df.groupby('Model', dropna=False)[metric].mean().sort_values(ascending=ascending)
    fig, ax = plt.subplots(figsize=(6,4))
    s.plot(kind='bar', ax=ax)
    ax.set_title(title or f'{metric} by model')
    ax.set_ylabel(metric)
    if ylim is not None:
        ax.set_ylim(*ylim)
    ax.grid(axis='y', alpha=0.3)
    annotate_bars(ax)
    plt.show()

11) Core Scores + Model Aggregation (Presentation-Ready)¶

Step 1 builds 3 core scores for each summary: Coverage, Faithfulness, and Framing (plus Similarity for context).

Step 2 aggregates those scores by model to create the main results table.

#Create 3 CORE SCORES

# Coverage Score
if 'coverage_score' not in df.columns and {'covered_points', 'total_key_points'}.issubset(df.columns):
    df['coverage_score'] = df['covered_points'] / df['total_key_points'].replace(0, np.nan)

# Faithfulness Score
if 'faithfulness_score' not in df.columns and 'contradiction_rate' in df.columns:
    df['faithfulness_score'] = 1 - df['contradiction_rate']
elif 'contradiction_rate' in df.columns:
    # Recompute to enforce the single-number definition consistently
    df['faithfulness_score'] = 1 - df['contradiction_rate']

# Framing Score
# Main framing signal: keyword overlap
df['framing_score'] = df['keyword_overlap']

# Optional alternative (kept for reference):
# df['framing_score'] = df['keyword_overlap'] - df['sentiment_shift'].abs()

# Keep a clean display table at summary-level
core_cols = [c for c in ['coverage_score', 'faithfulness_score', 'framing_score', 'similarity'] if c in df.columns]
display(df[core_cols].head())

# Step 2 — Aggregate by Model (main result)
model_col = 'model' if 'model' in df.columns else 'Model'

    df.groupby(model_col, dropna=False)[core_cols]

          'faithfulness_score': 'Faithfulness',
          'framing_score': 'Framing',
